This example demonstrates how rbx-tsx transforms React components, including useState, useCallback hooks, TypeScript interfaces for props, and JSX syntax.

<Tabs>
  <Tab title="Input (TypeScript)">
    ```tsx
    /**
     * Simple React component — useState, useCallback, props, JSX
     */
    
    import React, { useState, useCallback } from "react";
    
    interface CounterProps {
      label: string;
      initialCount?: number;
    }
    
    export default function Counter({ label, initialCount = 0 }: CounterProps) {
      const [count, setCount] = useState(initialCount);
    
      const increment = useCallback(() => {
        setCount((c) => c + 1);
      }, []);
    
      const decrement = useCallback(() => {
        setCount((c) => c - 1);
      }, []);
    
      return (
        <div className="counter">
          <h1>{label}</h1>
          <span>Count: {count}</span>
          <div className="button-row">
            <button className="primary" onClick={increment}>
              +
            </button>
            <button className="secondary" onClick={decrement}>
              -
            </button>
          </div>
        </div>
      );
    }
    ```
  </Tab>

<Note>
  Key transformations:
  - React imports are converted to require statements for Roblox packages
  - Interface types are preserved as Luau type definitions
  - Optional props with defaults use Luau's conditional expression (`if x ~= nil then x else default`)
  - JSX elements become `React.createElement` calls
  - HTML elements like `<div>` map to Roblox UI classes (Frame, TextButton)
  - `className` becomes `[React.Tag]` and `onClick` becomes `[React.Event.Activated]`
</Note>
