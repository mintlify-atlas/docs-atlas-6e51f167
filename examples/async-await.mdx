---
title: Async/Await
description: Transform async functions into Promise-based Luau code
---

This example shows how rbx-tsx transforms async/await syntax into Promise-based Luau code using the Promise library.

<Tabs>
  <Tab title="Input (TypeScript)">
    ```typescript
    /**
     * Async/await patterns — compiles to Promise.new + :expect()
     */
    
    async function fetchConfig(): Promise<{ timeout: number }> {
      return Promise.resolve({ timeout: 5000 });
    }
    
    async function loadData(): Promise<string> {
      const config = await fetchConfig();
      return `Loaded with timeout ${config.timeout}`;
    }
    
    async function main() {
      try {
        const result = await loadData();
        console.log(result);
      } catch (err) {
        console.error("Failed:", err);
      }
    }
    
    main();
    ```
  </Tab>
  <Tab title="Output (Luau)">
    ```lua
    -- Auto-generated by rbx-tsx
    -- Source: async-await.ts
    
    local Promise = require(game:GetService("ReplicatedStorage").Packages.Promise)
    -- 
    -- *
    --  * Async/await patterns — compiles to Promise.new + :expect()
    local function fetchConfig(): any
        return Promise.new(function(resolve, reject)
            resolve(Promise:resolve({ timeout = 5000 }))
            return
        end)
    end
    local function loadData(): any
        return Promise.new(function(resolve, reject)
            local config = fetchConfig():expect()
            resolve(`Loaded with timeout {config.timeout}`)
            return
        end)
    end
    local function main()
        return Promise.new(function(resolve, reject)
            local _ok, err = pcall(function()
                local result = loadData():expect()
                print(result)
            end)
            if not _ok then
                warn("[ERROR]", "Failed:", err)
            end
        end)
    end
    main()
    ```
  </Tab>
</Tabs>

<Note>
  Key transformations:
  - `async` functions are wrapped in `Promise.new(function(resolve, reject) ... end)`
  - `await` expressions are transformed to `:expect()` calls on Promises
  - `try/catch` blocks become `pcall()` with error handling
  - `console.error` maps to `warn("[ERROR]", ...)`
  - String template literals are converted to Luau interpolated strings
</Note>
