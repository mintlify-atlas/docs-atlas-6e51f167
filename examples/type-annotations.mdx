---
title: Type Annotations
description: Transform TypeScript type system features to Luau types
---

This example demonstrates how rbx-tsx preserves TypeScript type information in Luau, including type aliases, interfaces, generics, and union types.

<Tabs>
  <Tab title="Input (TypeScript)">
    ```typescript
    /**
     * Type aliases, interfaces, unions, and generics
     */
    
    type ID = string | number;
    
    interface BaseEntity {
      id: ID;
      createdAt: number;
    }
    
    interface Player extends BaseEntity {
      name: string;
      health: number;
    }
    
    type Rarity = "common" | "uncommon" | "rare" | "legendary";
    
    interface Item {
      id: string;
      name: string;
      rarity: Rarity;
    }
    
    // Generic function
    function getById<T extends BaseEntity>(entities: T[], id: ID): T | undefined {
      return entities.find((e) => e.id === id);
    }
    
    // Union type usage
    function formatId(id: ID): string {
      return typeof id === "string" ? id : String(id);
    }
    
    const players: Player[] = [
      { id: "1", name: "Player1", health: 100, createdAt: 0 },
    ];
    
    const found = getById(players, "1");
    if (found) {
      console.log(found.name, found.health);
    }
    ```
  </Tab>
  <Tab title="Output (Luau)">
    ```lua
    -- Auto-generated by rbx-tsx
    -- Source: types.ts
    
    -- 
    local function _arrayFind(t, fn)
        for i, v in t do
            if fn(v, i) then
                return v
            end
        end
        return nil
    end
    -- 
    -- Type aliases, interfaces, unions, and generics
    type ID = string | number
    type BaseEntity = {
        id: ID,
        createdAt: number,
    }
    type Player = {
        name: string,
        health: number,
    }
    type Rarity = "common" | "uncommon" | "rare" | "legendary"
    type Item = {
        id: string,
        name: string,
        rarity: Rarity,
    }
    -- Generic function
    local function getById<T>(entities: { T }, id: ID): T?
        return _arrayFind(entities, function(e) return e.id == id end)
    end
    -- Union type usage
    local function formatId(id: ID): string
        return if typeof(id) == "string" then id else tostring(id)
    end
    local players: { Player } = { { id = "1", name = "Player1", health = 100, createdAt = 0 } }
    local found = getById(players, "1")
    if found then
        print(found.name, found.health)
    end
    ```
  </Tab>
</Tabs>

<Note>
  Key transformations:
  - Type aliases and interfaces are preserved as Luau `type` definitions
  - Union types (`string | number`) are supported in Luau with the same syntax
  - String literal unions (`"common" | "uncommon"`) are preserved
  - Generics are maintained with `<T>` syntax
  - Array types `T[]` become `{ T }` in Luau
  - Optional types `T | undefined` become `T?`
  - Interface extension is flattened into the type definition
  - `.find()` is replaced with a helper function `_arrayFind`
  - `typeof` checks and `String()` conversions are adapted to Luau equivalents
</Note>
