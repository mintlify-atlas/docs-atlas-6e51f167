---
title: Decorators
description: Transform TypeScript decorators into Luau wrapper functions
---

This example demonstrates how rbx-tsx transforms class and method decorators into Luau code.

<Tabs>
  <Tab title="Input (TypeScript)">
    ```typescript
    /**
     * Decorators — class decorators as wrapper calls, method decorators as descriptor modifications
     */
    
    // Class decorator: receives constructor, can return replacement
    function sealed(target: any) {
      return target;
    }
    
    // Method decorator: receives (target, propertyKey, descriptor), can return modified descriptor
    function log(target: any, key: string, desc: any) {
      return desc;
    }
    
    @sealed
    class Greeter {
      name: string;
    
      constructor(name: string) {
        this.name = name;
      }
    
      @log
      greet(): string {
        return `Hello, ${this.name}`;
      }
    }
    
    const g = new Greeter("World");
    print(g.greet());
    ```
  </Tab>
  <Tab title="Output (Luau)">
    ```lua
    -- Auto-generated by rbx-tsx
    -- Source: decorators.ts
    
    -- *
    --  * Decorators — class decorators as wrapper calls, method decorators as descriptor modifications
    -- Class decorator: receives constructor, can return replacement
    local function sealed(target: any)
        return target
    end
    -- Method decorator: receives (target, propertyKey, descriptor), can return modified descriptor
    local function log(target: any, key: string, desc: any)
        return desc
    end
    local Greeter = {}
    Greeter.__index = Greeter
    type GreeterData = {
        name: string,
    }
    type Greeter = typeof(setmetatable({} :: GreeterData, Greeter))
    function Greeter.new(name: string): Greeter
        local self = {}
        self.name = name
        return setmetatable(self, Greeter)
    end
    local _opt0 = function(self: Greeter): string return `Hello, {self.name}` end
    local _opt1 = { value = _opt0 }
    _opt1 = log(Greeter, "greet", _opt1) or _opt1
    Greeter.greet = if _opt1 and _opt1.value then _opt1.value else _opt0
    local _opt2 = Greeter
    _opt2 = sealed(_opt2) or _opt2
    Greeter = _opt2
    local g = Greeter.new("World")
    print(g:greet())
    ```
  </Tab>
</Tabs>

<Note>
  Key transformations:
  - Classes become Luau tables with metatables and `__index` metamethod
  - Constructors become `.new()` static methods
  - Method decorators create descriptor objects `{ value = fn }` that are passed to the decorator function
  - Class decorators are applied after the class definition by calling the decorator and optionally replacing the class
  - `new ClassName()` syntax becomes `ClassName.new()`
</Note>
