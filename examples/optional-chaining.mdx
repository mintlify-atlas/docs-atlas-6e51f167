---
title: Optional Chaining
description: Transform optional chaining and nullish coalescing operators
---

This example demonstrates how rbx-tsx transforms optional chaining (`?.`) and nullish coalescing (`??`) operators into Luau conditional expressions.

<Tabs>
  <Tab title="Input (TypeScript)">
    ```typescript
    /**
     * Optional chaining (?.) and nullish coalescing (??)
     */
    
    interface User {
      name: string;
      profile?: {
        avatar?: string;
        settings?: {
          theme?: string;
        };
      };
    }
    
    const user: User | null = {
      name: "Player",
      profile: {
        avatar: "rbxassetid://123",
        settings: { theme: "dark" },
      },
    };
    
    // Optional chaining — short-circuits to undefined if any part is null/undefined
    const theme = user?.profile?.settings?.theme;
    const missing = user?.profile?.missing?.nested;
    
    // Nullish coalescing — use default only for null/undefined (not 0 or "")
    const displayTheme = theme ?? "light";
    const displayName = user?.name ?? "Anonymous";
    
    // Optional call
    type Callback = (() => void) | undefined;
    const cb: Callback = undefined;
    cb?.();
    
    console.log(displayTheme, displayName);
    ```
  </Tab>
  <Tab title="Output (Luau)">
    ```lua
    -- Auto-generated by rbx-tsx
    -- Source: optional-chaining.ts
    
    -- *
    --  * Optional chaining (?.) and nullish coalescing (??)
    type User = {
        name: string,
        profile: { avatar: string?, settings: { theme: string? }? }?,
    }
    local user: User? = {
        name = "Player",
        profile = { avatar = "rbxassetid://123", settings = { theme = "dark" } },
    }
    -- Optional chaining — short-circuits to undefined if any part is null/undefined
    local _opt0 = if user ~= nil then user.profile else nil
    local _opt1 = if _opt0 ~= nil then _opt0.settings else nil
    local theme = if _opt1 ~= nil then _opt1.theme else nil
    local _opt2 = if user ~= nil then user.profile else nil
    local _opt3 = if _opt2 ~= nil then _opt2.missing else nil
    local missing = if _opt3 ~= nil then _opt3.nested else nil
    -- Nullish coalescing — use default only for null/undefined (not 0 or "")
    local displayTheme = if theme ~= nil then theme else "light"
    local displayName = if (if user ~= nil then user.name else nil) ~= nil then if user ~= nil then user.name else nil else "Anonymous"
    -- Optional call
    type Callback = ((() -> ()))?
    local cb: Callback = nil
    if cb ~= nil then
        cb()
    end
    print(displayTheme, displayName)
    ```
  </Tab>
</Tabs>

<Note>
  Key transformations:
  - `a?.b?.c` chains become nested conditional expressions: `if a ~= nil then a.b else nil`, then `if _opt0 ~= nil then _opt0.c else nil`
  - `x ?? y` becomes `if x ~= nil then x else y`
  - Optional method calls `fn?.()` become `if fn ~= nil then fn() end`
  - Each step in a chain is assigned to a temporary variable (`_opt0`, `_opt1`, etc.) for clarity
</Note>
